Defines TypeScript interfaces and type guards for handling message data structures in both GraphQL and core application formats within the Flamingo/OpenFrame chat system.

## Key Components

### Types & Interfaces
- **`GraphQLMessage`** - Server-side message format with dialog metadata, chat type, and owner information
- **`CoreMessage`** - Client-side message format with role-based structure for UI rendering
- **`Message`** - Type alias for CoreMessage (primary message type used in components)
- **`MessageConnection`** - GraphQL connection pattern for paginated message queries
- **`MessagesResponse`** - API response wrapper for message queries
- **`MessagePage`** - Processed page of messages with pagination metadata

### Type Guards
- **`isGraphQLMessage()`** - Runtime check to identify server-format messages
- **`isCoreMessage()`** - Runtime check to identify client-format messages

## Usage Example

```typescript
import { Message, isGraphQLMessage, isCoreMessage } from './message.types'

// Handle mixed message formats
function processMessage(message: unknown): Message | null {
  if (isGraphQLMessage(message)) {
    // Transform GraphQL format to CoreMessage
    return {
      id: message.id,
      role: message.owner.type === 'USER' ? 'user' : 'assistant',
      content: message.messageData,
      timestamp: new Date(message.createdAt)
    }
  }
  
  if (isCoreMessage(message)) {
    return message // Already in correct format
  }
  
  return null
}

// Paginate through messages
function handleMessagePage(response: MessagesResponse) {
  const { messages, pageInfo } = response.data.messages
  
  if (pageInfo.hasNextPage) {
    // Load more messages using endCursor
    loadMessages(pageInfo.endCursor)
  }
}
```
