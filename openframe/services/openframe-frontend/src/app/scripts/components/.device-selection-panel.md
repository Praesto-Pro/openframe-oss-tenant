A React component that provides a comprehensive interface for searching, filtering, and selecting devices from a list with organization-based filtering and bulk selection capabilities.

## Key Components

- **DeviceSelectionPanel** - Main component that renders search, filter, and device selection interface
- **DeviceSelectionPanelProps** - Interface defining all props including device data, loading states, search/filter controls, and selection handlers
- **OrganizationOption** - Type for organization filter dropdown options

## Usage Example

```typescript
import { DeviceSelectionPanel } from './device-selection-panel'

function DeviceManager() {
  const [devices, setDevices] = useState<Device[]>([])
  const [selectedIds, setSelectedIds] = useState<Set<string>>(new Set())
  const [searchTerm, setSearchTerm] = useState('')
  const [selectedOrgIds, setSelectedOrgIds] = useState<string[]>([])
  
  const organizationOptions = [
    { label: 'Engineering', value: 'eng-001' },
    { label: 'Marketing', value: 'mkt-001' }
  ]

  const handleToggleSelect = (id: string) => {
    const newSelection = new Set(selectedIds)
    if (newSelection.has(id)) {
      newSelection.delete(id)
    } else {
      newSelection.add(id)
    }
    setSelectedIds(newSelection)
  }

  return (
    <DeviceSelectionPanel
      devices={devices}
      isLoading={false}
      searchTerm={searchTerm}
      onSearchChange={setSearchTerm}
      organizationOptions={organizationOptions}
      selectedOrgIds={selectedOrgIds}
      onOrgIdsChange={setSelectedOrgIds}
      selectedIds={selectedIds}
      onToggleSelect={handleToggleSelect}
      onSelectAll={() => setSelectedIds(new Set(devices.map(getDevicePrimaryId)))}
      onClearSelection={() => setSelectedIds(new Set())}
      selectedCount={selectedIds.size}
    />
  )
}
```

The component features responsive design with separate search and organization filter sections, bulk selection controls, and a grid layout that adapts from single column on mobile to four columns on extra-large screens.
