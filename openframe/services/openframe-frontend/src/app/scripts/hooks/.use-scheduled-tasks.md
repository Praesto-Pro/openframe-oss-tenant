A React hook for managing scheduled tasks in the Tactical RMM platform, providing query and mutation capabilities for tasks associated with specific scripts.

## Key Components

- **ScheduledTask**: Complete interface for scheduled task objects with timing, actions, and configuration
- **ScheduledTaskAction**: Interface for script-based actions within scheduled tasks  
- **scheduledTasksQueryKeys**: Query key factory for React Query cache management
- **useScheduledTasks**: Main hook providing scheduled task operations for a given script
- **fetchScheduledTasksByScript**: API function to retrieve tasks filtered by script ID
- **deleteScheduledTaskApi**: API function for task deletion

## Usage Example

```typescript
import { useScheduledTasks } from './use-scheduled-tasks'

function ScriptScheduledTasks({ scriptId }: { scriptId: string }) {
  const {
    scheduledTasks,
    isLoading,
    error,
    deleteTask,
    isDeleting,
    refetch
  } = useScheduledTasks(scriptId)

  const handleDeleteTask = (taskId: string) => {
    deleteTask(taskId, {
      onSuccess: () => console.log('Task deleted successfully'),
      onError: (error) => console.error('Failed to delete:', error.message)
    })
  }

  if (isLoading) return <div>Loading scheduled tasks...</div>
  if (error) return <div>Error: {error}</div>

  return (
    <div>
      {scheduledTasks.map(task => (
        <div key={task.id}>
          <h3>{task.name}</h3>
          <p>Type: {task.task_type}</p>
          <p>Enabled: {task.enabled ? 'Yes' : 'No'}</p>
          <button 
            onClick={() => handleDeleteTask(task.id.toString())}
            disabled={isDeleting}
          >
            Delete Task
          </button>
        </div>
      ))}
    </div>
  )
}
```
