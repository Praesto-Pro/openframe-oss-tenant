A React hook for managing approval request actions with built-in loading states and error handling. It provides methods to approve or reject requests while handling async operations through React Query mutations.

## Key Components

- **`ApprovalRequestAction`** - Type definition for approval request parameters
- **`approveRequest()`** - API function that sends approval/rejection requests to the backend
- **`useApprovalRequests()`** - Main hook that provides approval management functionality
- **`handleApproveRequest()`** - Method to approve a request with optional callbacks
- **`handleRejectRequest()`** - Method to reject a request with optional callbacks

## Usage Example

```typescript
import { useApprovalRequests } from '@/hooks/use-approval-requests'
import { APPROVAL_STATUS } from '@/constants'

function ApprovalButton({ requestId }: { requestId: string }) {
  const { 
    handleApproveRequest, 
    handleRejectRequest, 
    isLoading, 
    error 
  } = useApprovalRequests()

  const onApprove = async () => {
    await handleApproveRequest(requestId, {
      onSuccess: (status) => {
        console.log(`Request ${status}`)
        // Handle success (e.g., show toast, update UI)
      },
      onError: (error) => {
        console.error('Approval failed:', error.message)
      }
    })
  }

  const onReject = async () => {
    await handleRejectRequest(requestId, {
      onSuccess: () => console.log('Request rejected'),
      onError: (error) => console.error('Rejection failed:', error)
    })
  }

  return (
    <div>
      <button onClick={onApprove} disabled={isLoading}>
        {isLoading ? 'Processing...' : 'Approve'}
      </button>
      <button onClick={onReject} disabled={isLoading}>
        Reject
      </button>
      {error && <p className="error">{error}</p>}
    </div>
  )
}
```
